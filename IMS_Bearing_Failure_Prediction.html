Conditioning monitoring¶

This exercise demonstrates the use of machine learning to identify a normal operational threshold to prevent failures.

There are two approaches:

  • comparing the Mahalanobis distance from each data point to the normal group centroid
  • comparing the mean absolute error from the error loss from each data point to the predicted data point using a Neural Network Autoencoder

# Covariance using numpy cov
def cov_matrix(data, verbose=False):
    covariance_matrix = np.cov(data, rowvar=False)
    if is_pos_def(covariance_matrix):
        inv_covariance_matrix = np.linalg.inv(covariance_matrix)
        if is_pos_def(inv_covariance_matrix):
            return covariance_matrix, inv_covariance_matrix
        else:
            print("Error: Inverse of Covariance Matrix is not positive definite!")
    else:
        print("Error: Covariance Matrix is not positive definite!")

def cov_matrix_EmpiricalCovariance(data):
    cov_matrix = EmpiricalCovariance().fit(data)

    cov_inv = np.linalg.inv(cov)
    return cov, cov_inv
In [12]:
# Mahalanobis distance
def MahalanobisDist(inv_cov_matrix, mean_distr, data, verbose=False):
    inv_covariance_matrix = inv_cov_matrix
    vars_mean = mean_distr
    diff = data - vars_mean
    md = []
    for i in range(len(diff)):
        md.append(np.sqrt(diff[i].dot(inv_covariance_matrix).dot(diff[i])))
    return md
In [13]:
# Detect outliers using a threshold value of 3 times the std dev
def MD_detectOutliers(dist, extreme=False, verbose=False):
    k = 3. if extreme else 2.
    threshold = np.mean(dist) * k
    outliers = []
    for i in range(len(dist)):
        if dist[i] >= threshold:
            outliers.append(i)  # index of the outlier
    return np.array(outliers)

In [15]:
# Check if positive definitive
def is_pos_def(A):
    if np.allclose(A, A.T):
        try:
            np.linalg.cholesky(A)
            return True
        except np.linalg.LinAlgError:
            return False
    else:
        return False
